# COMANDOS
git --version ---Verificar si Git está instalado.
git config --global user.name "Tu Nombre Apellido" --- Configurar nombre de usuario.
git config --global user.email "tu_email@dominio.com" ---Configurar correo electrónico.
cd <ruta> ---Cambiar de directorio.
mkdir Repo1 ---Crear carpeta llamada Repo1.
cd Repo1 ---Entrar a la carpeta Repo1.
git init ---Inicializar un repositorio Git en la carpeta actual.
git status --- Ver estado del repositorio (archivos rastreados/no rastreados).
git add README.md ---Añadir archivo específico al área de preparación.
git add . ---Añadir todos los archivos nuevos/modificados al área de preparación.
git commit -m "Mensaje del commit" ---Guardar cambios con un mensaje descriptivo.
git log --oneline ---Ver historial de commits en una sola línea por cada uno.
